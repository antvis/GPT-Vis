Creating a Chart

import { GPTVis } from '@antv/gpt-vis';

const vis = new GPTVis({
  container: '#container',
  width: 600,
  height: 400,
  withContainer: true
});

// Render with syntax
await vis.render(`vis pie\ndata\n  - category A\n    value 10`);

// Or render with JSON
await vis.render({
  type: 'pie',
  data: [{ category: 'A', value: 10 }]
});
